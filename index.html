3: Find unique values in array?

                                    const array = [1,2,3,3,4,4,5,6,6,6];

                                    const uniqueArray = [];
                                    
                                    for(i=0; i < array.length; i++) {
                                        let count = 0;
                                        for(j=0; j < array.length; j++) {
                                            if(array[i] == array[j]) {
                                                count++
                                            }
                                        }
                                        if (count === 1) {
                                            uniqueArray.push(array[i])
                                        }
                                    }
                
                                    console.log(uniqueArray);
                
                                    //Output: [1, 2, 5]
4: Find min and max values in array?

                                    const array = [15,86,9,63,151,5];
                
                                    let minValue = Math.min(...array);
                
                                    let maxValue = Math.max(...array);
                
                                    let minVal = array.reduce((prev,curr) => {
                                        return prev < curr ? prev : curr;
                                    })
                
                                    let maxVal = array.reduce((prev,curr) => {
                                        return prev > curr ? prev : curr;
                                    })
                
                                    console.log("Math Min:" + minValue);
                                    console.log("Math Max:" + maxValue);
                                    console.log("Reduce Min:" + minVal);
                                    console.log("Reduce Max:" + maxVal);
                
                                    //Output 
                                    //Math Min:5
                                    //Math Max:151
                                    //Reduce Min:5
                                    //Reduce Max:151
